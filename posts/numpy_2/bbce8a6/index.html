多维度聚合

按行or列进行聚合

M = np.random.random((3,4))
print(M)
[[0.02467873 0.06724963 0.67939277 0.45369684]
 [0.53657921 0.89667129 0.99033895 0.21689698]
 [0.6630782  0.26332238 0.020651   0.75837865]]

M.sum()
5.570934646992556

#指定轴方向进行聚合
M.min(axis=0)
#得到四列数据
array([0.02467873, 0.06724963, 0.020651  , 0.21689698])

#找到最大一行数据
M.max(axis=1)
array([0.67939277, 0.99033895, 0.75837865])
